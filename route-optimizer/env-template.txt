# 1. Get your API key from: https://console.cloud.google.com/google/maps-apis/

# 3. Copy your API key below
# 4. Create a file named ".env.local" in the project root
# 5. Add the line below to that file
# 6. Restart the development server: npm run dev

NEXT_PUBLIC_GOOGLE_MAPS_API_KEY=your_api_key_here
